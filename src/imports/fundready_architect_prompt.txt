The Business Success Scan (BSS) collects 10 hard data points across 3 steps:
  BSS Step 1: Business name, EIN, contact, address, phone
  BSS Step 2: Entity type, start date, industry, revenue, bank account,
              bank age, avg balance, NSF history, CC sales, A/R, equipment
  BSS Step 3: Credit scores (3 bureaus), utilization, derogatory marks,
              bankruptcy, inquiries, new accounts, personal income

The FundScore™ Assessment asks 24 questions.

  Q0  → Personal credit score        (BSS Step 3: Experian/TU/Equifax sliders)
  Q1  → Credit utilization            (BSS Step 3: utilization slider)
  Q2  → Derogatory marks             (BSS Step 3: derogatory checkboxes)

  Q12 → NSF/overdraft count          (BSS Step 2: NSF field)
  Q13 → Average daily balance        (BSS Step 2: avg balance cards)
  Q14 → Funds separated?             (BSS Step 2: bank account type)
  Q16 → Entity type                  (BSS Step 2: entity type cards)
  Q17 → Years in business            (BSS Step 2: start date → calculated)
  Q18 → Industry                     (BSS Step 2: industry dropdown)

fix the problem — it papers over it while creating a second source of truth,

  hasTaxLiens:     boolean;
  noDerogItems:    boolean;
  inquiries30d:    '0' | '1_2' | '3_4' | '5plus' | 'unknown';
  bizCreditFile:   'paydex_80plus' | 'below_80' | 'building' | 'none';

  // Part 2 — Readiness (indices 0–13 for READINESS_QUESTIONS array)
  readinessAnswers: (number | undefined)[];
}

const WEIGHTS = { C: 0.28, D: 0.22, F: 0.20, B: 0.13, S: 0.10, N: 0.07 };

export function computeScore(data: UnifiedAnswers): {
  score: number;
  dimAvg: Record<'C'|'D'|'F'|'B'|'S'|'N', number>;
  bankableScore: number;
  napScore: number;
} {
  const buckets: Record<string, number[]> = { C:[], D:[], F:[], B:[], S:[], N:[] };
  let totalBoost = 0;

  // ── PART 1 — FOUNDATION DATA → DIMENSION BUCKETS ─────────────────────

  // C: Personal FICO
  const composite = [data.experian, data.transunion, data.equifax].sort((a,b)=>a-b)[1];
  buckets.C.push(
    composite < 580 ? 0.05 : composite < 620 ? 0.20 : composite < 650 ? 0.40 :
    composite < 700 ? 0.60 : composite < 740 ? 0.78 : 1.00
  );

  // C: Utilization
  buckets.C.push(
    data.utilization < 20 ? 1.00 : data.utilization < 30 ? 0.75 :
    data.utilization < 50 ? 0.45 : 0.10
  );

  // C: Derogatories
  const derogScore = data.noDerogItems ? 1.00 :
    (data.hasCollections || data.hasChargeoffs || data.hasTaxLiens) ?
    (data.hasBankruptcy && data.bankruptcyAge === 'under_2yr' ? 0.10 : 0.20) :
    data.hasLatePay ? 0.40 : 0.60;
  buckets.C.push(derogScore);

  // F: Monthly revenue
  buckets.F.push(
    data.monthlyRevenue < 3000  ? 0.10 : data.monthlyRevenue < 8000  ? 0.30 :
    data.monthlyRevenue < 20000 ? 0.55 : data.monthlyRevenue < 50000 ? 0.78 : 1.00
  );

  // B: Fund separation
  buckets.B.push(
    data.bankAccount === 'dedicated' ? 1.00 : data.bankAccount === 'personal' ? 0.50 : 0.00
  );

  // B: Average daily balance
  buckets.B.push(
    data.avgDailyBalance === 'near_zero' ? 0.00 : data.avgDailyBalance === '500_2k' ? 0.25 :
    data.avgDailyBalance === '2k_10k' ? 0.55 : data.avgDailyBalance === '10k_25k' ? 0.80 : 1.00
  );

  // B: NSF count
  buckets.B.push(
    data.nsfCount === 'zero' ? 1.00 : data.nsfCount === '1_2' ? 0.65 :
    data.nsfCount === '3_5' ? 0.25 : 0.00
  );

  // S: Entity type
  buckets.S.push(
    data.entityType === 'sole_prop' ? 0.10 : data.entityType === 'llc_single' ? 0.65 :
    data.entityType === 'llc_multi' ? 0.78 : 1.00
  );

  // S: Business age (calculated from startDate)
  const ageMonths = (new Date().getFullYear() - data.startDate.year) * 12
    + (new Date().getMonth() - data.startDate.month + 1);
  buckets.S.push(
    ageMonths < 6  ? 0.05 : ageMonths < 12 ? 0.25 : ageMonths < 24 ? 0.55 :
    ageMonths < 60 ? 0.85 : 1.00
  );

  // S: Industry
  const industryScores = {
    professional_tech_health: 1.00, retail_ecomm: 0.85, construction_re: 0.65,
    restaurant_food: 0.40, transport: 0.50, other: 0.70
  };
  buckets.S.push(industryScores[data.industry] || 0.70);

  // ── PART 2 — READINESS QUESTIONS → DIMENSION BUCKETS ─────────────────

  READINESS_QUESTIONS.forEach((q, qi) => {
    const ai = data.readinessAnswers[qi];
    if (ai === undefined) return;
    const opt = q.options[ai];
    if (!opt?.score) return;
    if (opt.boost) totalBoost += opt.boost;
    Object.entries(opt.score).forEach(([dim, val]) => {
      if (buckets[dim]) buckets[dim].push(val as number);
    });
  });

  // ── COMPUTE DIMENSION AVERAGES ─────────────────────────────────────────
  const dimAvg: Record<string, number> = {};
  Object.keys(buckets).forEach(d => {
    const vals = buckets[d];
    dimAvg[d] = vals.length ? vals.reduce((a,b) => a+b, 0) / vals.length : 0.50;
  });

  // ── WEIGHTED COMPOSITE → 0–1000 ────────────────────────────────────────
  let base = 0;
  Object.entries(WEIGHTS).forEach(([d, w]) => { base += (dimAvg[d] || 0) * w; });
  const score = Math.max(0, Math.min(1000,
    Math.round(base * 1000 + Math.min(totalBoost, 80))
  ));

  // ── BANKABLE SCORE (from Part 1 hard data) ─────────────────────────────
  const bankableScore = calculateBankableScore(data);

  // ── NAP SCORE ──────────────────────────────────────────────────────────
  const napScore = calculateNAPScore(data);

  return { score, dimAvg: dimAvg as Record<'C'|'D'|'F'|'B'|'S'|'N', number>, bankableScore, napScore };
}

function getHardRequirements(product, data: UnifiedAnswers): {
  passed: string[];
  failed: string[];
  status: 'pass' | 'fail';
} {
  const checks: { label: string; pass: boolean }[] = [];

  // Universal checks:
  checks.push({ label: `FICO ${product.minFICO}+ required`, pass: composite >= product.minFICO });
  checks.push({ label: `Business ${product.minAge}+ months`, pass: ageMonths >= product.minAge });
  checks.push({ label: `Revenue $${product.minRevenue}/mo`, pass: data.monthlyRevenue >= product.minRevenue });

  // Product-specific:
  if (product.requiresEIN)      checks.push({ label: 'EIN required', pass: data.hasEIN });
  if (product.requiresDedBank)  checks.push({ label: 'Dedicated bank account', pass: data.bankAccount === 'dedicated' });
  if (product.requiresNoBkrupt) checks.push({ label: 'No recent bankruptcy', pass: !data.hasBankruptcy || data.bankruptcyAge === 'over_7yr' });
  if (product.requiresNoJudg)   checks.push({ label: 'No judgments/liens', pass: !data.hasJudgments });
  if (product.requiresCCSales)  checks.push({ label: `$${product.minCCSales}/mo CC sales`, pass: data.ccSales >= product.minCCSales });
  if (product.requiresAR)       checks.push({ label: 'Active accounts receivable', pass: data.arBalance > 0 });
  if (product.requiresEquip)    checks.push({ label: 'Equipment to finance', pass: data.equipmentValue > 0 });
  if (product.requiresPO)       checks.push({ label: `POs worth $${product.minPO}+`, pass: data.poBalance >= product.minPO });
  if (product.requiresProperty) checks.push({ label: 'Investment property owned', pass: data.ownsProperty === 'yes' });
  if (product.requiresRental)   checks.push({ label: 'Rental income present', pass: data.propertyData?.rentalIncome > 0 });
  if (product.requiresConstruct)checks.push({ label: 'Construction project planned', pass: data.constructionPlan === 'yes' });
  if (product.requiresIncome75k)checks.push({ label: '$75K+ personal income', pass: ['75_125k','125_250k','over_250k'].includes(data.personalIncome) });

  const allPassed = checks.every(c => c.pass);
  return {
    passed: checks.filter(c=>c.pass).map(c=>c.label),
    failed: checks.filter(c=>!c.pass).map(c=>c.label),
    status: allPassed ? 'pass' : 'fail'
  };
}

function getStatus(product, currentScore, compDone, data) {
  const hard = getHardRequirements(product, data);
  const scoreMet = currentScore >= product.minScore;
  const compMet = product.compReq.every(id => compDone.has(id));

  if (scoreMet && compMet && hard.status === 'pass') return 'pre_qualified';

  const scoreClose = currentScore >= product.minScore - 80;
  const compClose = product.compReq.filter(id => !compDone.has(id)).length <= 3;
  const hardClose = hard.failed.length <= 2;

  if (scoreClose && compClose && hard.status === 'pass') return 'approaching';
  if (scoreMet && compMet && hardClose) return 'likely_qualified';

  return 'not_pre_qualified';
}
